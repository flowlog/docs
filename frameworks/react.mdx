## Setup

<InstallFlowlogSDK />

Then initialize the Flowlog SDK in your app:

<CodeGroup>

```js src/index.js
// src/index.js
import React from "react";
import ReactDOM from "react-dom/client";
import App from "./App";
import flowlog from "flowlog-js";
import { FlowlogProvider } from "flowlog-js/react";

const flowlowInstance = flowlog({
  apiKey: "API_TOKEN",
});

const root = ReactDOM.createRoot(document.getElementById("root"));

root.render(
  <React.StrictMode>
    <FlowlogProvider client={flowlowInstance}>
      <App />
    </FlowlogProvider>
  </React.StrictMode>
);
```

</CodeGroup>

### Track your first event

<CodeGroup>

```jsx src/track-button.js
import { useFlowlog } from "flowlog-js/react";

export default function Index() {
  const flowlog = useFlowlog();

  return (
    <button
      onClick={() => {
        flowlog?.track("button_clicked", { button_name: "home_button" });
      }}
    >
      Click here
    </button>
  );
}
```

</CodeGroup>

### Optional: Track page views

You can track page views with `react-router-dom` with `useLocation` hook:

<CodeGroup>

```jsx src/App.js
import { Fragment, useEffect } from "react";
import { useFlowlog } from "flowlog-js/react";
import { Route, Routes, Link, useLocation } from "react-router-dom";

function App() {
  const flowlog = useFlowlog();
  const location = useLocation();

  useEffect(() => {
    flowlog?.page();
  }, [location, flowlog]);

  return (
    <Fragment>
      <nav>
        <ul>
          <li>
            <Link to="/">Home</Link>
          </li>
          <li>
            <Link to="/about">About</Link>
          </li>
        </ul>
      </nav>
      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />
      </Routes>
    </Fragment>
  );
}

const Home = () => {
  return (
    <div>
      <h1>Home</h1>
    </div>
  );
};

const About = () => {
  return <h1>About</h1>;
};

export default App;
```

</CodeGroup>
