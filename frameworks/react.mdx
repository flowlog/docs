## Setup

<InstallFlowlogSDK />

Then initialize the Flowlog SDK in your app:

<CodeGroup>

```js src/index.js
// src/index.js
import React from "react";
import ReactDOM from "react-dom/client";
import App from "./App";
import flowlog from "flowlog-js";
import { FlowlogProvider } from "flowlog-js/react";

const flowlowInstance = flowlog({
  apiKey: "API_TOKEN",
});

const root = ReactDOM.createRoot(document.getElementById("root"));

root.render(
  <React.StrictMode>
    <FlowlogProvider client={flowlowInstance}>
      <App />
    </FlowlogProvider>
  </React.StrictMode>
);
```

</CodeGroup>

### Track your first event

<CodeGroup>

```jsx src/track-button.js
import { useFlowlog } from "flowlog-js/react";

export default function Index() {
  const flowlog = useFlowlog();

  return (
    <button
      onClick={() => {
        flowlog?.track("button_clicked", { button_name: "home_button" });
      }}
    >
      Click here
    </button>
  );
}
```

</CodeGroup>
