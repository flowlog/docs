### Create a 'src/flowlog-client' file to initialize the Flowlog SDK

<CodeGroup>

```jsx flowlog-client.ts
import flowlogJS from "flowlog-js";

export default flowlogJS({
  apiKey: "YOUR_API_KEY",
});
```

### Track your first event

<CodeGroup>

```js button-track.js
<script setup lang="ts">
import { ref } from "vue";
import flowlogjs from "./flowlog-client";

const count = ref(0);

function handleClick() {
  flowlogjs.track("click-button", {
    count: count.value,
  });
  count.value++;
}
</script>

<template>
  <button type="button" @click="handleClick">count is {{ count }}</button>
</template>
```

</CodeGroup>

### Track your first pageview

<Info>
  This example uses Vue Router to track page views.
</Info>

<CodeGroup>
```ts main.ts
import Vue from "vue";
import App from "./App.vue";
import VueRouter from "vue-router";
import flowlogjs from "./flowlog-client";
import routes from "./routes";


Vue.use(VueRouter);

const router = new VueRouter({
  mode: "history",
  routes,
});


router.afterEach((to, from) => {
  console.log(`Route change to ${to.path} from ${from.path}`);
  flowlogjs.page(); // Track page view
});

new Vue({
  router,
  render: (h) => h(App),
}).$mount("#app");

```

</CodeGroup>
